257#define JPC_COX_LAZY 0x01 /* Selective arithmetic coding bypass. */

756{
757 return !(x == JPC_MS_SOC || x == JPC_MS_SOD || x == JPC_MS_EOC ||
758 x == JPC_MS_EPH || (x >= 0xff30 && x <= 0xff3f));
759}
